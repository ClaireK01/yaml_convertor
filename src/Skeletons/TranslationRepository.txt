<?php

namespace App\Repository;

use App\Entity\{NAME};
use App\Entity\{NAME}Translation;
use Doctrine\Bundle\DoctrineBundle\Repository\ServiceEntityRepository;
use Doctrine\Persistence\ManagerRegistry;

/**
 * @extends ServiceEntityRepository<{NAME}Translation>
 *
 * @method {NAME}Translation|null find($id, $lockMode = null, $lockVersion = null)
 * @method {NAME}Translation|null findOneBy(array $criteria, array $orderBy = null)
 * @method {NAME}Translation[]    findAll()
 * @method {NAME}Translation[]    findBy(array $criteria, array $orderBy = null, $limit = null, $offset = null)
 */
class {NAME}TranslationRepository extends ServiceEntityRepository
{
    public function __construct(ManagerRegistry $registry)
    {
        parent::__construct($registry, {NAME}Translation::class);
    }

    public function add({NAME}Translation $entity, bool $flush = false): void
    {
        $this->getEntityManager()->persist($entity);

        if ($flush) {
            $this->getEntityManager()->flush();
        }
    }

    public function remove({NAME}Translation $entity, bool $flush = false): void
    {
        $this->getEntityManager()->remove($entity);

        if ($flush) {
            $this->getEntityManager()->flush();
        }
    }

}
